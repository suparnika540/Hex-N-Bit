distortions = []
K = range(1,10)
for k in K:
    kmeanModel = KMeans(n_clusters=k)  
    kmeanModel.fit(x)                  
    distortions.append(kmeanModel.inertia_)   
plt.plot(K, distortions, 'r*-')
plt.xlabel('Value of K')
plt.ylabel('Distortion')
plt.title('Elbow Plot')
